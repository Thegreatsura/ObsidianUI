# Stacked Logos

Multiple logo sets that animate in/out while stacked on top of each other with a mouse glow effect.

<ComponentPreview
  component={
    <div className="relative w-full h-[300px] rounded-lg overflow-hidden bg-white dark:bg-zinc-950 flex items-center justify-center">
      <StackedLogos logoGroups={demoLogoGroups} duration={10} />
    </div>
  }
  code={`import StackedLogos from '@/components/ui/stacked-logos'

const logoGroups = [
  [<img src="/logo1.svg" />, <img src="/logo2.svg" />],
  [<img src="/logo3.svg" />, <img src="/logo4.svg" />],
]

export function Demo() {
  return <StackedLogos logoGroups={logoGroups} />
}`}
/>
