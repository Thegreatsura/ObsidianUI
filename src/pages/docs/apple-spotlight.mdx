# Apple Spotlight

A macOS-style spotlight search component with animated shortcuts and search results. Features a beautiful blur effect, animated shortcut buttons, and live search results.

<ComponentPreview
  component={
    <div className="relative w-full rounded-lg overflow-hidden bg-black">
      <video autoPlay loop muted playsInline className="w-full h-auto max-h-[600px] object-cover">
        <source src="/demos/apple-spotlight.mp4" type="video/mp4" />
      </video>
    </div>
  }
  code={`import { AppleSpotlight } from '@/components/ui/apple-spotlight'
import { useState } from 'react'

export function Demo() {
  const [isOpen, setIsOpen] = useState(false)
  
  return (
    <>
      <button onClick={() => setIsOpen(true)}>Open Spotlight</button>
      <AppleSpotlight isOpen={isOpen} handleClose={() => setIsOpen(false)} />
    </>
  )
}`}
/>
