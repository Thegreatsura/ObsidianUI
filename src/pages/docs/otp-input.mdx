## Usage

<CodeBlock language="tsx" code={`import { OTPInput } from "@/components/ui/otp-input"

export function Demo() {
  return (
    <OTPInput
      correctOTP="424242"
      onSuccess={() => console.log("Verified!")}
      onError={() => console.log("Invalid OTP")}
    />
  )
}`} />
