# Icon Wheel

A beautiful rotating wheel of icons that spin smoothly around a central point. Perfect for showcasing tech stacks, skills, or brand logos.

<ComponentPreview
  component={
    <div className="relative w-full h-[420px] rounded-lg overflow-hidden bg-gradient-to-br from-slate-50 via-neutral-50 to-zinc-100 dark:from-zinc-900 dark:via-zinc-900 dark:to-zinc-950 flex items-center justify-center">
      <IconWheel icons={demoIcons} radius={130} iconSize={36} duration={30} />
    </div>
  }
  code={`import IconWheel from '@/components/ui/icon-wheel'

const icons = [
  "/icon1.svg",
  "/icon2.svg",
  "/icon3.svg",
  // ... more icons
]

export function Demo() {
  return <IconWheel icons={icons} radius={150} iconSize={48} duration={20} />
}`}
/>
