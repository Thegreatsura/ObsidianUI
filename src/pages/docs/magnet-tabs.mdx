# Magnet Tabs

Animated tab navigation with magnetic hover effect and smooth indicator transitions.

<ComponentPreview
  component={
    <div className="relative w-full h-[350px] rounded-lg overflow-hidden bg-white dark:bg-zinc-950">
      <video autoPlay loop muted playsInline className="w-full h-auto max-h-[600px] object-cover">
        <source src="/demos/magnet-tabs.mp4" type="video/mp4" />
      </video>
    </div>
  }
  code={`import { MagnetTabs } from '@/components/ui/magnet-tabs'
import { useState } from 'react'

export function Demo() {
  const [activeTab, setActiveTab] = useState("All")
  return (
    <MagnetTabs
      slug="demo"
      options={["All", "Latest", "Popular", "Featured"]}
      activeTab={activeTab}
      onSelect={setActiveTab}
    />
  )
}`}
/>
