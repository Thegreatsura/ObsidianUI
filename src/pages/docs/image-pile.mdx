# Image Pile

An auto-rotating stack of images that shuffle with smooth animations.

<ComponentPreview
  component={
    <div className="relative w-full rounded-lg overflow-hidden bg-black">
      <video autoPlay loop muted playsInline className="w-full h-auto max-h-[600px] object-cover">
        <source src="https://cdn.obsidianui.dev/demos/image-pile.mp4" type="video/mp4" />
      </video>
    </div>
  }
  code={`import { ImagePile } from '@/components/ui/image-pile'

export function Demo() {
  return (
    <ImagePile 
      images={["/image1.jpg", "/image2.jpg", "/image3.jpg"]} 
      speed={2} 
    />
  )
}`}
/>
