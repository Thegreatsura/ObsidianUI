# Masonry Grid

A responsive masonry-style image grid with hover animations and lightbox support.

<Dependencies step={2} title="Add util file">
    <p className="text-[13px] text-muted-foreground mt-2 mb-4">Create a file <code className="text-foreground font-mono text-xs bg-neutral-100 dark:bg-[#161616] border border-neutral-200 dark:border-neutral-800 px-2 py-1 rounded-md">lib/utils.ts</code>:</p>
    <CodeBlock language="tsx" code={`import { ClassValue, clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

}`} />
</Dependencies>
