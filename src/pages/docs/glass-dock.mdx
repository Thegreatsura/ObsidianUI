# Glass Dock

A macOS-style dock with glassmorphism effect, smooth magnification animations, and morphing SVG icons.

  component={

  code={`import { GlassDock } from '@/components/ui/glass-dock'
import { Home, Mail, Github, Linkedin, MapPin } from 'lucide-react'

const items = [
  { title: "Home", icon: Home, href: "#" },
  { title: "Email", icon: Mail, href: "#" },
  { title: "Marker", icon: MapPin, href: "#" },
  { title: "LinkedIn", icon: Linkedin, href: "#" },
  { title: "Github", icon: Github, href: "#" },
]

export function Demo() {
  return <GlassDock items={items} />
}`}
