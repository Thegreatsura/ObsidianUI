# Pixelated Carousel

An image carousel with a unique pixelated transition effect. Images dissolve into pixels and reassemble for the next slide.

<ComponentPreview
  component={
    <div className="relative w-full h-[400px] rounded-lg overflow-hidden bg-zinc-950">
      <PixelatedCarousel 
        images={[
          "https://images.unsplash.com/photo-1506905925346-21bda4d32df4?w=800&h=600&fit=crop",
          "https://images.unsplash.com/photo-1511300636408-a63a89df3482?w=800&h=600&fit=crop",
        ]}
        pixelSize={80}
      />
    </div>
  }
  code={`import { PixelatedCarousel } from '@/components/ui/pixelated-carousel'

export function Demo() {
  const images = [
    '/images/slide1.jpg',
    '/images/slide2.jpg',
    '/images/slide3.jpg',
  ]
  
  return <PixelatedCarousel images={images} pixelSize={100} />
}`}
/>
